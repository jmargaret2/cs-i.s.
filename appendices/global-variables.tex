		\hline

		T & period of a wave \\
		\multirow{2}{*}{$\omega$} & angular frequency: frequency of a sine or \\ & cosine wave as it moves counterclockwise around the unit circle \\

		y & displacement of a wave laterally \\
		$y_m$ & amplitude \\
		$\lambda$ & wavelength \\
		\hline
	\end{tabular}
\caption{Variables used in this paper}
